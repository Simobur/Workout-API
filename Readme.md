# Explore Rest APIs

The app defines following CRUD APIs.
/api/workout 

# Users:

    GET /users
    
    POST /users
    
    GET /users/{id}
    
    PUT /users/{id}
    
    DELETE /users/{id}

# Routines

    GET /routines

    POST /routines
    
    GET /routines/{id}
    
    PUT /routines/{id}
    
    DELETE /routines/{id}

    GET by UserId /routines/user?id="id"

# Exercises

    GET /exercises

    POST /exercises
    
    GET /exercises/{id}
    
    PUT /exercises/{id}
    
    DELETE /exercises/{id}

    GET by RoutineId /exercises/routine?id="id"
